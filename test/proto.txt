import "google/protobuf/descriptor.proto";

extend google.protobuf.MessageOptions {
  optional string int8 = int8;
  optional string char8 = char8;
  optional string int16 = int16;
  optional string uint8 = uint8;
  optional string byte = byte;
  optional string uint16 = uint16;
  
}

message MyMessage {
  option (my_option) = "Hello world!";
}

{ 
	required int64 secs = 1; 
	required int64 nsecs = 2; 
 } 

message LogItem
{ 
	required int64 seq = 1; 
	required TimeStamp timeStamp = 2; 
 }
